158    async def fetch(
159        self,
160        source_type: DataSourceType,
161        *args,
162        failover: bool = True,
163        health_aware: bool = True,
164        **kwargs
165    ) -> DataSourceResult:
166        """
167        获取数据（自动选择数据源）
168
169        Args:
170            source_type: 数据源类型
171            *args: 位置参数传递给数据源
172            failover: 是否启用故障切换
173            health_aware: 是否启用健康感知选择
174            **kwargs: 关键字参数传递给数据源
175
176        Returns:
177            DataSourceResult: 第一个成功的数据源结果
178        """
179        async with await self._get_semaphore():
180            sources = self._get_ordered_sources(source_type)
181            errors = []
182
183            # 如果启用健康感知，尝试获取最健康的数据源
184            if health_aware and failover:
185                healthy_source = await self._health_interceptor.get_healthy_source(
186                    sources, prefer_healthy=True
187                )
188                if healthy_source:
189                    # 将健康的数据源排在前面
190                    sources = [healthy_source] + [s for s in sources if s != healthy_source]
191
192            for source in sources:
193                if not self._source_configs.get(source.name, DataSourceConfig(
194                    source_class=type(source),
195                    name=source.name,
196                    source_type=source.source_type
197                )).enabled:
198                    continue
199
200                # 如果启用健康感知，跳过不健康的数据源
201                if health_aware and self._health_interceptor.should_skip_source(source.name):
202                    errors.append(f"{source.name}: 数据源不健康")
203                    continue
204
205                try:
206                    result = await source.fetch(*args, **kwargs)
207
208                    # 记录请求
209                    self._record_request(source.name, source_type, result)
210
211                    if result.success:
212                        # 更新健康检查状态
213                        if health_aware:
214                            await self._health_checker.check_source(source)
215                        return result
216
217                    errors.append(f"{source.name}: {result.error}")
218
219                except Exception as e:
220                    errors.append(f"{source.name}: {str(e)}")
221                    self._record_request(source.name, source_type, None, error=str(e))
222
223            # 所有数据源都失败
224            if failover and errors:
225                return DataSourceResult(
226                    success=False,
227                    error=f"所有数据源均失败: {'; '.join(errors)}",
228                    timestamp=time.time(),
229                    source="manager",
230                    metadata={"source_type": source_type.value, "errors": errors}
231                )
232
233            return DataSourceResult(
234                success=False,
235                error="没有可用的数据源",
236                timestamp=time.time(),
237                source="manager",
238                metadata={"source_type": source_type.value}
239            )

276    async def fetch_batch(
277        self,
278        source_type: DataSourceType,
279        params_list: list[dict[str, Any]],
280        *args,
281        parallel: bool = True,
282        **kwargs
283    ) -> list[DataSourceResult]:
284        """
285        批量获取数据
286
287        Args:
288            source_type: 数据源类型
289            params_list: 参数列表，每个元素包含 args 和 kwargs
290            *args: 通用位置参数
291            parallel: 是否并行执行请求
292            **kwargs: 通用关键字参数
293
294        Returns:
295            List[DataSourceResult]: 结果列表
296        """
297        sources = self._get_ordered_sources(source_type)
298        if not sources:
299            return [
300                DataSourceResult(
301                    success=False,
302                    error="没有可用的数据源",
303                    timestamp=time.time(),
304                    source="manager"
305                )
306            ] * len(params_list)
307
308        primary_source = sources[0]
309
310        if not parallel:
311            # 串行执行（兼容旧行为）
312            results = []
313            for params in params_list:
314                async with await self._get_semaphore():
315                    try:
316                        result = await primary_source.fetch(
317                            *params.get("args", args),
318                            **kwargs,
319                            **params.get("kwargs", {})
320                        )
321                        results.append(result)
322                    except Exception as e:
323                        results.append(
324                            DataSourceResult(
325                                success=False,
326                                error=str(e),
327                                timestamp=time.time(),
328                                source=primary_source.name
329                            )
330                        )
331            return results
332
333        # 并行执行 - 使用信号量限制并发数
334        async def fetch_one(params: dict[str, Any]) -> DataSourceResult:
335            async with await self._get_semaphore():
336                try:
337                    result = await primary_source.fetch(
338                        *params.get("args", args),
339                        **kwargs,
340                        **params.get("kwargs", {})
341                    )
342                    return result
343                except Exception as e:
344                    return DataSourceResult(
345                        success=False,
346                        error=str(e),
347                        timestamp=time.time(),
348                        source=primary_source.name
349                    )
350
351        # 使用 gather 并行执行所有请求
352        tasks = [fetch_one(params) for params in params_list]
353        results = await asyncio.gather(*tasks, return_exceptions=True)
354
355        # 处理异常情况
356        processed_results = []
357        for i, result in enumerate(results):
358            if isinstance(result, Exception):
359                processed_results.append(
360                    DataSourceResult(
361                        success=False,
362                        error=str(result),
363                        timestamp=time.time(),
364                        source=primary_source.name
365                    )
366                )
367            else:
368                processed_results.append(result)
369
370        return processed_results

372    def _get_ordered_sources(self, source_type: DataSourceType) -> list[DataSource]:
373        """
374        获取排序后的数据源列表
375
376        Returns:
377            List[DataSource]: 按优先级排序的数据源
378        """
379        source_ids = self._type_sources[source_type]
380
381        if not self._enable_load_balancing or len(source_ids) <= 1:
382            # 按优先级排序
383            def get_priority(sid):
384                config = self._source_configs.get(sid)
385                if config:
386                    return config.priority
387                return 0
388            return [
389                self._sources[sid]
390                for sid in sorted(source_ids, key=get_priority)
391            ]
392
393        # 负载均衡模式
394        idx = self._round_robin_index[source_type] % len(source_ids)
395        self._round_robin_index[source_type] += 1
396
397        # 轮询选择
398        ordered = source_ids[idx:] + source_ids[:idx]
399        return [self._sources[sid] for sid in ordered]

401    async def health_check(self, source_name: str | None = None) -> dict[str, Any]:
402        """
403        执行健康检查
404
405        Args:
406            source_name: 可选，指定检查的数据源
407
408        Returns:
409            Dict: 健康检查结果
410        """
411        if source_name:
412            source = self._sources.get(source_name)
413            if not source:
414                return {"status": "unknown", "error": f"数据源不存在: {source_name}"}
415
416            result = await self._health_checker.check_source(source)
417            return {
418                "source": source_name,
419                "status": result.status.value,
420                "response_time_ms": result.response_time_ms,
421                "error_count": result.error_count,
422                "last_check": result.last_check.isoformat(),
423                "message": result.message,
424                "details": result.details
425            }
426
427        # 检查所有数据源
428        sources = list(self._sources.values())
429        results = await self._health_checker.check_all_sources(sources)
430
431        healthy_count = sum(1 for r in results.values() if r.status == HealthStatus.HEALTHY)
432        degraded_count = sum(1 for r in results.values() if r.status == HealthStatus.DEGRADED)
433        unhealthy_count = sum(1 for r in results.values() if r.status == HealthStatus.UNHEALTHY)
434
435        return {
436            "total_sources": len(results),
437            "healthy_count": healthy_count,
438            "degraded_count": degraded_count,
439            "unhealthy_count": unhealthy_count,
440            "sources": {
441                name: result.to_dict()
442                for name, result in results.items()
443            }
444        }

572    def get_statistics(self) -> dict[str, Any]:
573        """
574        获取统计数据
575
576        Returns:
577            Dict: 统计数据
578        """
579        total_requests = len(self._request_history)
580        successful_requests = sum(1 for r in self._request_history if r["success"])
581        failed_requests = total_requests - successful_requests
582
583        # 按数据源统计
584        source_stats: dict[str, dict[str, Any]] = {}
585        for source_name in self._sources:
586            source_requests = [r for r in self._request_history if r["source"] == source_name]
587            if source_requests:
588                source_stats[source_name] = {
589                    "total": len(source_requests),
590                    "success": sum(1 for r in source_requests if r["success"]),
591                    "failed": sum(1 for r in source_requests if not r["success"]),
592                    "success_rate": sum(1 for r in source_requests if r["success"]) / len(source_requests)
593                }
594
595        return {
596            "total_requests": total_requests,
597            "successful_requests": successful_requests,
598            "failed_requests": failed_requests,
599            "overall_success_rate": successful_requests / total_requests if total_requests > 0 else 0,
600            "source_statistics": source_stats,
601            "registered_sources": {
602                name: {
603                    "type": s.source_type.value,
604                    "enabled": self._source_configs.get(name, DataSourceConfig(
605                        source_class=type(s),
606                        name=name,
607                        source_type=s.source_type
608                    )).enabled,
609                    "priority": self._source_configs.get(name, DataSourceConfig(
610                        source_class=type(s),
611                        name=name,
612                        source_type=s.source_type
613                    )).priority
614                }
615                for name, s in self._sources.items()
616            },
617            "max_concurrent": self._semaphore._value if hasattr(self._semaphore, '_value') else None
618        }

284    ) -> list[DataSourceResult]:
