 48class DataSourceHealthChecker:
 49    """
 50    数据源健康检查器
 51
 52    功能:
 53    - 定期检查数据源健康状态
 54    - 记录健康历史
 55    - 计算统计数据
 56    - 支持自动恢复检测
 57    """

 59    def __init__(
 60        self,
 61        check_interval: int = 60,
 62        timeout: float = 10.0,
 63        max_response_time_ms: float = 5000.0,
 64        consecutive_failure_threshold: int = 3
 65    ):
 66        """
 67        初始化健康检查器
 68
 69        Args:
 70            check_interval: 检查间隔（秒）
 71            timeout: 请求超时时间（秒）
 72            max_response_time_ms: 最大响应时间（毫秒），超过此值视为降级
 73            consecutive_failure_threshold: 连续失败次数阈值
 74        """
 75        self.check_interval = check_interval
 76        self.timeout = timeout
 77        self.max_response_time_ms = max_response_time_ms
 78        self.consecutive_failure_threshold = consecutive_failure_threshold
 79
 80        self.health_history: dict[str, list[HealthCheckResult]] = {}
 81        self._check_tasks: dict[str, asyncio.Task] = {}
 82        self._running = False
 83        self._consecutive_failures: dict[str, int] = {}

 85    async def check_source(self, source: DataSource, save_history: bool = True) -> HealthCheckResult:
 86        """
 87        检查单个数据源健康状态
 88
 89        Args:
 90            source: 数据源实例
 91            save_history: 是否保存到历史记录
 92
 93        Returns:
 94            HealthCheckResult: 健康检查结果
 95        """
 96        source_name = source.name
 97        start_time = time.perf_counter()
 98
 99        try:
100            # 执行健康检查
101            is_healthy = await asyncio.wait_for(
102                source.health_check(),
103                timeout=self.timeout
104            )
105
106            response_time_ms = (time.perf_counter() - start_time) * 1000
107
108            # 判断健康状态
109            if is_healthy:
110                # 重置连续失败计数（健康时重置）
111                self._consecutive_failures[source_name] = 0
112
113                if response_time_ms > self.max_response_time_ms:
114                    status = HealthStatus.DEGRADED
115                    message = f"响应时间过长: {response_time_ms:.2f}ms"
116                else:
117                    status = HealthStatus.HEALTHY
118                    message = "数据源健康"
119
120                result = HealthCheckResult(
121                    source_name=source_name,
122                    status=status,
123                    response_time_ms=response_time_ms,
124                    error_count=0,
125                    last_check=datetime.now(),
126                    message=message,
127                    details={"response_time_threshold": self.max_response_time_ms}
128                )
129            else:
130                self._consecutive_failures[source_name] = \
131                    self._consecutive_failures.get(source_name, 0) + 1
132
133                error_count = self._consecutive_failures[source_name]
134
135                if error_count >= self.consecutive_failure_threshold:
136                    status = HealthStatus.UNHEALTHY
137                    message = f"连续 {error_count} 次检查失败"
138                else:
139                    status = HealthStatus.DEGRADED
140                    message = f"检查失败 {error_count} 次"
141
142                result = HealthCheckResult(
143                    source_name=source_name,
144                    status=status,
145                    response_time_ms=response_time_ms,
146                    error_count=error_count,
147                    last_check=datetime.now(),
148                    message=message
149                )
150
151            # 保存到历史记录
152            if save_history:
153                if source_name not in self.health_history:
154                    self.health_history[source_name] = []
155                self.health_history[source_name].append(result)
156
157                # 限制历史记录数量
158                max_history = 100
159                if len(self.health_history[source_name]) > max_history:
160                    self.health_history[source_name] = self.health_history[source_name][-max_history:]
161
162            return result
163
164        except asyncio.TimeoutError:
165            response_time_ms = (time.perf_counter() - start_time) * 1000
166            self._consecutive_failures[source_name] = \
167                self._consecutive_failures.get(source_name, 0) + 1
168
169            result = HealthCheckResult(
170                source_name=source_name,
171                status=HealthStatus.UNHEALTHY,
172                response_time_ms=response_time_ms,
173                error_count=self._consecutive_failures[source_name],
174                last_check=datetime.now(),
175                message=f"健康检查超时（>{self.timeout}s）"
176            )
177
178            # 保存到历史记录
179            if save_history:
180                if source_name not in self.health_history:
181                    self.health_history[source_name] = []
182                self.health_history[source_name].append(result)
183
184            return result
185
186        except Exception as e:
187            response_time_ms = (time.perf_counter() - start_time) * 1000
188            self._consecutive_failures[source_name] = \
189                self._consecutive_failures.get(source_name, 0) + 1
190
191            result = HealthCheckResult(
192                source_name=source_name,
193                status=HealthStatus.UNHEALTHY,
194                response_time_ms=response_time_ms,
195                error_count=self._consecutive_failures[source_name],
196                last_check=datetime.now(),
197                message=f"健康检查异常: {str(e)}",
198                details={"error_type": type(e).__name__}
199            )
200
201            # 保存到历史记录
202            if save_history:
203                if source_name not in self.health_history:
204                    self.health_history[source_name] = []
205                self.health_history[source_name].append(result)
206
207            return result

209    async def check_all_sources(
210        self,
211        sources: list[DataSource]
212    ) -> dict[str, HealthCheckResult]:
213        """
214        并行检查所有数据源
215
216        Args:
217            sources: 数据源列表
218
219        Returns:
220            Dict[str, HealthCheckResult]: 数据源名称到检查结果的映射
221        """
222        results = {}
223
224        # 并行执行检查（每个检查会自动保存到历史记录）
225        async def check_with_source(source: DataSource) -> tuple[str, HealthCheckResult]:
226            result = await self.check_source(source, save_history=False)  # 避免重复保存
227            return (source.name, result)
228
229        tasks = [check_with_source(source) for source in sources]
230        checked_results = await asyncio.gather(*tasks, return_exceptions=True)
231
232        for item in checked_results:
233            if isinstance(item, Exception):
234                # 如果出现异常，记录错误
235                continue
236
237            source_name, result = item
238            results[source_name] = result
239
240        return results

278    def get_statistics(self) -> dict[str, Any]:
279        """
280        获取健康检查统计数据
281
282        Returns:
283            Dict: 统计数据字典
284        """
285        stats = {
286            "total_sources_checked": len(self.health_history),
287            "sources": {}
288        }
289
290        for source_name, history in self.health_history.items():
291            if not history:
292                continue
293
294            recent_results = [r for r in history if r.status == HealthStatus.HEALTHY]
295            avg_response_time = sum(r.response_time_ms for r in history) / len(history)
296
297            stats["sources"][source_name] = {
298                "total_checks": len(history),
299                "healthy_count": len(recent_results),
300                "unhealthy_count": len(history) - len(recent_results),
301                "avg_response_time_ms": avg_response_time,
302                "current_status": history[-1].status.value,
303                "last_check": history[-1].last_check.isoformat()
304            }
305
306        return stats

337    async def start_background_check(
338        self,
339        sources: list[DataSource]
340    ):
341        """
342        启动后台健康检查任务
343
344        Args:
345            sources: 要监控的数据源列表
346        """
347        if self._running:
348            return
349
350        self._running = True
351
352        async def periodic_check():
353            while self._running:
354                await self.check_all_sources(sources)
355                await asyncio.sleep(self.check_interval)
356
357        self._check_task = asyncio.create_task(periodic_check())
358
359    def stop_background_check(self):
360        """停止后台健康检查任务"""
361        self._running = False
362        if hasattr(self, '_check_task'):
363            self._check_task.cancel()
364            try:
365                # 在事件循环中等待任务取消
366                loop = asyncio.get_event_loop()
367                loop.run_until_complete(self._check_task)
368            except asyncio.CancelledError:
369                pass
370
371    def reset(self):
372        """重置所有健康检查状态"""
373        self.health_history.clear()
374        self._consecutive_failures.clear()
375        self._running = False

378class HealthCheckInterceptor:
379    """
380    健康检查拦截器

383    """

385    def __init__(self, checker: DataSourceHealthChecker):
386        """
387        初始化拦截器
388
389        Args:
390            checker: 健康检查器实例
391        """
392        self.checker = checker
393
394    async def get_healthy_source(
395        self,
396        sources: list[DataSource],
397        prefer_healthy: bool = True
398    ) -> DataSource | None:
399        """
400        获取最健康的数据源
401
402        Args:
403            sources: 数据源列表
404            prefer_healthy: 是否优先选择健康的数据源
405
406        Returns:
407            DataSource: 选中的数据源，无健康数据源返回 None
408        """
409        if not sources:
410            return None
411
412        if not prefer_healthy:
413            return sources[0]
414
415        # 检查所有数据源
416        results = await self.checker.check_all_sources(sources)
417
418        # 优先选择健康的数据源
419        healthy_sources = [
420            s for s in sources
421            if results.get(s.name, HealthCheckResult(
422                source_name=s.name,
423                status=HealthStatus.UNKNOWN,
424                response_time_ms=0,
425                error_count=0,
426                last_check=datetime.now()
427            )).status == HealthStatus.HEALTHY
428        ]
429
430        if healthy_sources:
431            # 返回响应时间最短的健康数据源
432            return min(
433                healthy_sources,
434                key=lambda s: results.get(s.name, HealthCheckResult(
435                    source_name=s.name,
436                    status=HealthStatus.HEALTHY,
437                    response_time_ms=float('inf'),
438                    error_count=0,
439                    last_check=datetime.now()
440                )).response_time_ms
441            )
442
443        # 没有健康数据源，检查是否有降级的
444        degraded_sources = [
445            s for s in sources
446            if results.get(s.name, HealthCheckResult(
447                source_name=s.name,
448                status=HealthStatus.UNKNOWN,
449                response_time_ms=0,
450                error_count=0,
451                last_check=datetime.now()
452            )).status == HealthStatus.DEGRADED
453        ]
454
455        if degraded_sources:
456            return degraded_sources[0]
457
458        return None

460    def should_skip_source(self, source_name: str) -> bool:
461        """
462        判断是否应该跳过某个数据源
463
464        Args:
465            source_name: 数据源名称
466
467        Returns:
468            bool: True 表示应该跳过
469        """

472            return False

441                )).response_time_ms
